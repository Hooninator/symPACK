\section{finding indinstinguishable nodes when computing reachable set}

\begin{algorithm}
\DontPrintSemicolon
\KwIn{ $v$ is the node eliminated at step $s$. \reach{v} is its reachable set. $marker$ and $label$ arrays, $tag$}
$I \gets \emptyset$\;
$D \gets \emptyset$\;

$label(v) = s$\;

\ForAll{node $t \in \reach{v}$}{
  $tag = tag+1$\;
  \ForAll{node $u \in \reach{v}$}{
    $marker(u) = tag$\;
  }

  $tag = tag+1$\;
  $indist,\reachb{t} \gets get\_reach(t,tag,s,marker,label)$\;
  
  $deg(t) = \| \reachb{t} \|$\;

  \If{indist}{
    $I \gets I \cup {t}$\;
  }
  else{
    $D \gets D \cup {t}$\;
  }

}

\ForAll{node $t \in D$}{
  $deg(t) = deg(t) - \| I \|$\;
}

\ForAll{node $t \in I$}{
  $s = s+1$\;
  $label(t) = s$\;
}


\caption{Sketch of the MDO algorithm calling $get\_reach$}
\end{algorithm}
